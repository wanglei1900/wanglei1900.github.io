3. 字段详解 ​

字段描述示例
data:有效载荷（必须），可多行data: hello\n
id:事件ID，用于断线重连id: 42\n
event:事件类型（默认 "message"）event: status-update\n
retry:重连间隔（毫秒），建议设置retry: 5000\n
:注释行，客户端忽略: this is a comment\n

1. 基础API使用 ​

js
// 创建EventSource连接
const eventSource = new EventSource('/sse-endpoint');

// 监听消息事件（默认）
eventSource.onmessage = (event) => {
  console.log('Message:', event.data);
};

// 监听自定义事件
eventSource.addEventListener('status', (event) => {
  const data = JSON.parse(event.data);
  updateStatus(data.progress);
});

// 错误处理
eventSource.onerror = (error) => {
  console.error('SSE Error:', error);
  // 自动重连逻辑已内置
};

// 关闭连接
function closeConnection() {
  eventSource.close();
}

四、服务端实现 ​

1. Node.js 实现（Express） ​

js
const express = require('express');
const app = express();

app.get('/sse', (req, res) => {
  // 设置SSE响应头
  res.setHeader('Content-Type', 'text/event-stream');
  res.setHeader('Cache-Control', 'no-cache');
  res.setHeader('Connection', 'keep-alive');
  
  // 发送初始消息
  res.write('event: connected\n');
  res.write('data: Welcome to SSE server\n\n');
  
  // 发送定期更新
  let counter = 0;
  const intervalId = setInterval(() => {
    counter++;
    res.write(`id: ${Date.now()}\n`);
    res.write(`event: update\n`);
    res.write(`data: ${JSON.stringify({ count: counter, time: new Date() })}\n\n`);
    
    // 10次后停止
    if(counter >= 10) {
      clearInterval(intervalId);
      res.write('event: end\n');
      res.write('data: Stream ended\n\n');
      res.end();
    }
  }, 1000);
  
  // 客户端断开连接时清理
  req.on('close', () => {
    clearInterval(intervalId);
    res.end();
  });
});

app.listen(3000, () => console.log('SSE server running on port 3000'));

2. Python 实现（Flask） ​

python
from flask import Flask, Response, request
import time
import json

app = Flask(__name__)

@app.route('/stream')
def stream():
    def event_stream():
        client_id = request.args.get('client_id')
        # 发送初始消息
        yield f"event: connected\ndata: Client {client_id} connected\n\n"
        
        count = 0
        while True:
            count += 1
            # 构建事件数据
            data = {
                "timestamp": time.time(),
                "count": count,
                "client": client_id
            }
            yield f"id: {int(time.time()*1000)}\nevent: update\ndata: {json.dumps(data)}\n\n"
            time.sleep(1)
    
    return Response(event_stream(), mimetype="text/event-stream")

if __name__ == '__main__':
    app.run(threaded=True)

js
// 1. 认证机制
app.get('/secure-sse', (req, res) => {
  const token = req.headers['authorization'];
  if(!validateToken(token)) {
    res.status(401).end();
    return;

  message: "Too many SSE connections from this IP"
});
app.use('/sse', sseLimiter);

3. 性能优化 ​

js
// 1. 消息压缩
res.write(compressMessage(`data: ${largePayload}\n\n`));

// 2. 心跳包（保持连接活跃）
setInterval(() => {
  res.write(': heartbeat\n\n'); // 注释行不触发客户端事件
}, 30000);

// 3. 连接池管理
const activeConnections = new Set();

app.get('/sse', (req, res) => {
  activeConnections.add(res);
  req.on('close', () => activeConnections.delete(res));
});

// 广播消息函数
function broadcast(message) {
  activeConnections.forEach(res => {
    try {
      res.write(`data: ${message}\n\n`);
    } catch (e) {
      // 处理断开连接
      activeConnections.delete(res);
    }
  });
}

js
// 服务端
const stockPrices = {
  AAPL: 150.32,
  GOOGL: 2785.43,
  MSFT: 336.78
};

function simulatePriceChanges() {
  Object.keys(stockPrices).forEach(symbol => {
    const change = (Math.random() - 0.5) * 2; // -1% 到 +1%
    stockPrices[symbol] *= (1 + change/100);
    broadcast({
      type: 'price-update',
      symbol,
      price: stockPrices[symbol].toFixed(2),
      timestamp: Date.now()
    });
  });
}

setInterval(simulatePriceChanges, 1000);

// 客户端
const stockSource = new EventSource('/stocks');
stockSource.addEventListener('price-update', event => {
  const data = JSON.parse(event.data);
  updateStockTable(data.symbol, data.price);
});

2. 实时日志监控 ​

js
// 服务端
const { exec } = require('child_process');

app.get('/logs', (req, res) => {
  res.setHeader('Content-Type', 'text/event-stream');
  
  // 跟踪日志文件
  const tail = exec('tail -f /var/log/app.log');
  
  tail.stdout.on('data', data => {
    const lines = data.split('\n');
    lines.forEach(line => {
      if(line.trim()) {
        res.write(`data: ${JSON.stringify({ log: line })}\n\n`);
      }
    });
  });
  
  req.on('close', () => {
    tail.kill(); // 客户端断开时停止跟踪
  });
});

🛠️ 十、总结与最佳实践 ​

  • 核心优势：

    • 简单易实现，无需复杂协议
    • 自动重连机制提升可靠性
    • 天然支持HTTP基础设施（缓存、认证等）

    • 监控连接健康：定期心跳检测
    • 安全加固：验证来源、使用HTTPS、身份认
